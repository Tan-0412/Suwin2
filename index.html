// ════════════════════════════════════════════
//  ระบบจองรถ · Google Apps Script v1
//  Spreadsheet: Booking
// ════════════════════════════════════════════
const SHEETS = {
  customer: 'Customer',
  sc:       'SC',
  isuzu:    'ISUZU',
  status:   'STATUS',
  source:   'SOURCE',
  target:   'TARGET',
};

function doGet(e) {
  const p = e.parameter;
  try {
    switch(p.action) {
      case 'ping':        return ok({ pong: true });
      case 'getMeta':     return ok(getMeta());
      case 'getBookings': return ok({ status:'ok', data: getBookings(p) });
      case 'addBooking':  return ok(addBooking(p));
      case 'updateBooking': return ok(updateBooking(p));
      case 'deleteBooking': return ok(deleteBooking(p));
      case 'getTargets':  return ok({ status:'ok', data: getTargets() });
      case 'saveTarget':  return ok(saveTarget(p));
      case 'deleteTarget':return ok(deleteTarget(p));
      default: return ok({ status:'ready' });
    }
  } catch(err) { return ok({ status:'error', message: err.message }); }
}

function ss() { return SpreadsheetApp.getActiveSpreadsheet(); }

function getSheet(name) { return ss().getSheetByName(name); }

function sheetData(name) {
  const sh = getSheet(name);
  if (!sh) return [];
  const rows = sh.getDataRange().getValues();
  if (rows.length < 2) return [];
  const hdrs = rows[0].map(h => String(h).trim());
  return rows.slice(1).map(r => {
    const o = {};
    hdrs.forEach((h,i) => { o[h] = r[i] !== null && r[i] !== undefined ? String(r[i]).trim() : ''; });
    return o;
  }).filter(r => Object.values(r).some(v => v !== ''));
}

function getMeta() {
  const sc     = sheetData(SHEETS.sc).sort((a,b) => parseInt(a['ลำดับ']||99)-parseInt(b['ลำดับ']||99));
  const isuzu  = sheetData(SHEETS.isuzu).sort((a,b) => parseInt(a['ลำดับ']||99)-parseInt(b['ลำดับ']||99));
  const status = sheetData(SHEETS.status).sort((a,b) => parseInt(a['ลำดับ']||99)-parseInt(b['ลำดับ']||99));
  const source = sheetData(SHEETS.source).sort((a,b) => parseInt(a['ลำดับ']||99)-parseInt(b['ลำดับ']||99));
  return { sc, isuzu, status, source };
}

function getBookings(p) {
  const sh = getSheet(SHEETS.customer);
  if (!sh) return [];
  const rows = sh.getDataRange().getValues();
  if (rows.length < 2) return [];
  const hdrs = rows[0].map(h => String(h).trim());
  let data = rows.slice(1).map((r, i) => {
    const o = { _row: i + 2 };
    hdrs.forEach((h, j) => {
      const v = r[j];
      if (v instanceof Date) {
        o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'dd/MM/yyyy');
      } else {
        o[h] = v !== null && v !== undefined ? String(v).trim() : '';
      }
    });
    return o;
  }).filter(r => r['ชื่อลูกค้า'] || r['วันที่จอง']);

  if (p.sc)     data = data.filter(r => r['ที่ปรึกษาการขาย'] === p.sc);
  if (p.status) data = data.filter(r => r['สถานะ'] === p.status);
  if (p.month) {
    data = data.filter(r => {
      const d = r['วันที่จอง'];
      if (!d) return false;
      const parts = d.split('/');
      if (parts.length >= 3) return `${parts[2]}-${parts[1].padStart(2,'0')}` === p.month;
      return false;
    });
  }
  if (p.dateFrom) {
    data = data.filter(r => {
      const d = parseDate(r['วันที่จอง']);
      return d && d >= new Date(p.dateFrom);
    });
  }
  if (p.dateTo) {
    data = data.filter(r => {
      const d = parseDate(r['วันที่จอง']);
      return d && d <= new Date(p.dateTo + 'T23:59:59');
    });
  }
  // เรียงจากวันที่จองล่าสุดก่อน
  data.sort((a, b) => {
    const da = parseDate(a['วันที่จอง']);
    const db = parseDate(b['วันที่จอง']);
    if (!da && !db) return 0;
    if (!da) return 1;
    if (!db) return -1;
    return db - da;
  });
  return data;
}

function parseDate(s) {
  if (!s) return null;
  const p = String(s).split('/');
  if (p.length >= 3) return new Date(`${p[2]}-${p[1].padStart(2,'0')}-${p[0].padStart(2,'0')}`);
  return null;
}

function addBooking(p) {
  const sh = getSheet(SHEETS.customer);
  if (!sh) return { ok:false, error:'ไม่พบชีท Customer' };
  const hdrs = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(h=>String(h).trim());
  const COLS = ['วันที่จอง','ชื่อลูกค้า','เบอร์','รายละเอียดรถ','รุ่นรถ','สถานะ','วันที่ปล่อย','ที่ปรึกษาการขาย','ที่มาลูกค้า'];
  const newRow = Array(Math.max(hdrs.length, COLS.length)).fill('');
  COLS.forEach(col => {
    const idx = hdrs.indexOf(col);
    if (idx >= 0 && p[col] !== undefined) newRow[idx] = p[col];
  });
  sh.appendRow(newRow);
  return { ok:true, row: sh.getLastRow() };
}

function updateBooking(p) {
  const sh = getSheet(SHEETS.customer);
  if (!sh) return { ok:false, error:'ไม่พบชีท Customer' };
  const rowNum = parseInt(p._row);
  if (!rowNum || rowNum < 2) return { ok:false, error:'row ไม่ถูกต้อง' };
  const hdrs = sh.getRange(1,1,1,sh.getLastColumn()).getValues()[0].map(h=>String(h).trim());
  const existing = sh.getRange(rowNum,1,1,hdrs.length).getValues()[0];
  const COLS = ['วันที่จอง','ชื่อลูกค้า','เบอร์','รายละเอียดรถ','รุ่นรถ','สถานะ','วันที่ปล่อย','ที่ปรึกษาการขาย','ที่มาลูกค้า'];
  COLS.forEach(col => {
    const idx = hdrs.indexOf(col);
    if (idx >= 0 && p[col] !== undefined) existing[idx] = p[col];
  });
  sh.getRange(rowNum,1,1,hdrs.length).setValues([existing]);
  return { ok:true };
}

function deleteBooking(p) {
  const sh = getSheet(SHEETS.customer);
  if (!sh) return { ok:false, error:'ไม่พบชีท Customer' };
  const rowNum = parseInt(p._row);
  if (!rowNum || rowNum < 2) return { ok:false, error:'row ไม่ถูกต้อง' };
  sh.deleteRow(rowNum);
  return { ok:true };
}

function getTargets() {
  return sheetData(SHEETS.target);
}

function saveTarget(p) {
  const sh = getSheet(SHEETS.target);
  if (!sh) return { ok:false, error:'ไม่พบชีท TARGET' };
  const rows = sh.getDataRange().getValues();
  const hdrs = rows[0].map(h=>String(h).trim());
  const mi = hdrs.indexOf('เดือน');
  const ti = hdrs.indexOf('เป้าการจำหน่าย');
  // ถ้ามีเดือนนี้แล้ว update ถ้าไม่มีให้ append
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][mi]).trim() === p['เดือน']) {
      sh.getRange(i+1, ti+1).setValue(parseInt(p['เป้าการจำหน่าย'])||0);
      return { ok:true };
    }
  }
  sh.appendRow([p['เดือน'], parseInt(p['เป้าการจำหน่าย'])||0]);
  return { ok:true };
}

function deleteTarget(p) {
  const sh = getSheet(SHEETS.target);
  if (!sh) return { ok:false, error:'ไม่พบชีท TARGET' };
  const rowNum = parseInt(p._row);
  if (!rowNum || rowNum < 2) return { ok:false, error:'row ไม่ถูกต้อง' };
  sh.deleteRow(rowNum);
  return { ok:true };
}

function ok(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
